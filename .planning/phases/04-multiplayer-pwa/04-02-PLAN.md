    ```typescript
    import type { Move } from '../engine';

    export type P2PMessage =
      | { type: 'MOVE'; move: Move }
      | { type: 'GAME_SYNC'; moveHistory: Move[] }
      | { type: 'END_CHAIN' }
      | { type: 'PING' }
      | { type: 'PONG' };

    export function encodeMessage(msg: P2PMessage): string {
      return JSON.stringify(msg);
    }

    export function decodeMessage(data: string): P2PMessage {
      return JSON.parse(data) as P2PMessage;
    }
    ```

    Create src/multiplayer/webrtc.ts implementing the WebRTC connection lifecycle per research Pattern 1:
    - ICE_SERVERS: use Google's free STUN servers (`stun:stun.l.google.com:19302`, `stun:stun1.l.google.com:19302`).
    - `createOffer()` -- creates RTCPeerConnection, creates data channel ('game', ordered: true), creates offer, waits for ICE gathering with 15s timeout, returns Base64-encoded full SDP as offerCode + applyAnswer function.
    - `joinWithOffer(offerCode)` -- creates RTCPeerConnection, decodes and sets remote description, creates answer, waits for ICE gathering with 15s timeout, returns Base64-encoded answer code + P2PConnection. CRITICAL: register `ondatachannel` handler BEFORE calling `setRemoteDescription` to avoid race condition (Pitfall 2 from research).
    - `P2PConnection` interface: `{ send, close, onMessage, onStateChange }`.
    - Helper `wrapChannel(pc, channel)`: returns P2PConnection object wrapping the data channel.
    - ICE gathering timeout: if gathering doesn't complete in 15s, use whatever candidates are available. If zero candidates gathered, throw an error.
