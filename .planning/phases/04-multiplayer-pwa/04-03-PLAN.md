    - Add lint step before build: `npm run lint` (ensure package.json has a lint script; if not, add `"lint": "eslint src/"` to scripts).
    - Add test step before build: `npx vitest run` (runs full test suite, exits non-zero on failure).
    - Keep existing deploy job structure.
    - Final workflow order: checkout -> setup-node -> npm ci -> lint -> test -> build -> upload artifact -> deploy.

    Updated deploy.yml:
    ```yaml
    name: Deploy to GitHub Pages

    on:
      push:
        branches: [main]
      workflow_dispatch:

    permissions:
      contents: read
      pages: write
      id-token: write

    concurrency:
      group: pages
      cancel-in-progress: true

    jobs:
      build:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: 20
              cache: npm
          - run: npm ci
          - run: npm run lint
          - run: npx vitest run
          - run: npm run build
          - uses: actions/upload-pages-artifact@v4
            with:
              path: dist

      deploy:
        needs: build
        runs-on: ubuntu-latest
        environment:
          name: github-pages
          url: ${{ steps.deployment.outputs.page_url }}
        steps:
          - id: deployment
            uses: actions/deploy-pages@v4
    ```

    **Verify package.json scripts:**
