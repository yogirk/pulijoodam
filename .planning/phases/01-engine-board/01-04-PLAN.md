Helper for setting up board states in tests:
```typescript
// Utility: apply a sequence of moves and return final state
function playMoves(moves: Move[]): GameState {
  let state = createGame();
  for (const move of moves) {
    const result = applyMove(state, move);
    if (result.error) throw new Error(`Unexpected move error: ${result.error} for ${JSON.stringify(move)}`);
    state = result.state;
  }
  return state;
}

// Utility: mutate a GameState field for test setup (bypass move rules)
// Used for win/draw detection tests where reaching the state via legal moves is impractical
function withState(overrides: Partial<GameState>): GameState {
  return { ...createGame(), ...overrides };
}
```
