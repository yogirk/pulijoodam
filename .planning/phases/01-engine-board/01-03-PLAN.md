import type { GameState, Move, LegalMove, GameEvent, GameStatus, Piece } from '../engine';

interface UIState {
  gameState: GameState;
  history: GameState[];   // for undo — full state snapshots (UI layer concern)
  redoStack: GameState[]; // for redo
  selectedNode: number | null;
  legalMoves: LegalMove[];
  lastEvents: GameEvent[];
}

  | { type: 'NEW_GAME' }
  | { type: 'UNDO' }
  | { type: 'REDO' }
  | { type: 'END_CHAIN' }; // tiger voluntarily ends chain-hop

  <files>src/hooks/useGame.ts</files>
  <action>
    Create src/hooks/useGame.ts implementing the UIState / UIAction reducer described in the interfaces block.

    REDUCER LOGIC for NODE_TAPPED:
    1. If game is over (getGameStatus(state) !== 'ongoing'), ignore tap.
    2. If chainJumpInProgress !== null: tap on a legal capture destination executes the capture. Tap on the tiger itself has no effect. Other taps are ignored.
    3. If phase === 'placement' AND currentTurn === 'goat': attempt PLACE move. If move.error, no state change. On success, push old state to history, clear redoStack.
    4. If selectedNode === null: check if tapped node has a piece belonging to current player. If yes and getLegalMoves has moves from this node (for movement phase), set selectedNode. If no legal moves from this piece, no selection.
    5. If selectedNode !== null: check if tapped node is a legal destination. If yes, dispatch the move (MOVE or CAPTURE). Clear selection. Push history. If tapped the already-selected node or a non-destination, deselect.
    6. For tiger placement-phase moves: tap the tiger piece to select, then tap destination (empty adjacent) or tap over a goat to capture.

    REDUCER LOGIC for END_CHAIN: dispatch END_CHAIN move to engine. Push history. Clear selection.

    REDUCER LOGIC for UNDO: use undo() from engine. If history has a previous state, restore it. Set selectedNode = null, legalMoves = [], push current to redoStack.

    REDUCER LOGIC for REDO: use redo() from engine. If redoStack non-empty, restore. Clear selection.

    REDUCER LOGIC for NEW_GAME: reset to createGame(). Clear history, redoStack, selection.

    Export the hook:
    ```typescript
    export function useGame() {
      const [state, dispatch] = useReducer(gameReducer, initialUIState);
      return {
        gameState: state.gameState,
        selectedNode: state.selectedNode,
        legalMoves: state.legalMoves,
        lastEvents: state.lastEvents,
        status: getGameStatus(state.gameState),
        canUndo: state.history.length > 1,
        canRedo: state.redoStack.length > 0,
        onNodeTap: (nodeId: number) => dispatch({ type: 'NODE_TAPPED', nodeId }),
        onEndChain: () => dispatch({ type: 'END_CHAIN' }),
        onUndo: () => dispatch({ type: 'UNDO' }),
        onRedo: () => dispatch({ type: 'REDO' }),
        onNewGame: () => dispatch({ type: 'NEW_GAME' }),
      };
    }
    ```

    Note on getLegalMoves for selection: During movement phase, filter getLegalMoves(state) to moves where move.from === tappedNode. During placement phase for tigers, call getLegalMoves(state) filtered to moves involving the tapped tiger node as from.
