export type Piece = 'tiger' | 'goat';
export type Phase = 'placement' | 'movement';
export type Role = 'tiger' | 'goat';
export type GameStatus = 'ongoing' | 'tiger-wins' | 'goat-wins' | 'draw-repetition' | 'draw-50moves';

export interface GameState {
  board: (Piece | null)[];       // length 23; index = node id
  phase: Phase;
  currentTurn: Role;
  tigersInPool: number;
  goatsInPool: number;
  goatsCaptured: number;
  moveHistory: Move[];
  stateHashes: Record<string, number>;
  capturelessMoves: number;
  chainJumpInProgress: number | null;
  config: GameConfig;
}

export type Move =
  | { type: 'PLACE'; to: number }
  | { type: 'MOVE'; from: number; to: number }
  | { type: 'CAPTURE'; from: number; over: number; to: number }
  | { type: 'END_CHAIN' };

export interface LegalMove {
  move: Move;
  from?: number;
  to?: number;
}

export interface MoveResult {
  state: GameState;
  events: GameEvent[];
  error?: string;
}

Create `src/engine/ai/worker.ts` — the Web Worker entry point:
```typescript
import { chooseMove } from './index';
import type { AIRequest, AIResponse } from './types';

self.onmessage = (e: MessageEvent<AIRequest>) => {
  const { state, config } = e.data;
  const startTime = performance.now();
  const move = chooseMove(state, config);
  const elapsed = performance.now() - startTime;
  self.postMessage({
    type: 'MOVE_COMPUTED',
    move,
    thinkTimeMs: elapsed,
  } satisfies AIResponse);
};
```
